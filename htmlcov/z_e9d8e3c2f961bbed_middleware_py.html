15LEGACY_VIEW_MAP = { 

16 CourseExamsView: CourseExamsLegacyView,

18 ProctoringSettingsView: ProctoringSettingsLegacyView,

22class ExamRequestMiddleware(MiddlewareMixin): 

23 """ 

24 Intercept requests and determine if exams for this course 

25 should be managed by legacy edx-proctoring system. If so, requests 

26 are intercepted and forwarded to that service 

27 """ 

28 def process_view(self, request, view_func, view_args, view_kwargs): # pylint: disable=missing-function-docstring 

29 try: 

30 legacy_view = LEGACY_VIEW_MAP.get(view_func.view_class) 

31 except AttributeError: # pragma: no cover 

32 legacy_view = None 

33 

34 # call into override if a function is defined for this request method 

35 if legacy_view and getattr(legacy_view, request.method.lower()): 

36 course_id = view_kwargs.get('course_id') 

37 course_configuration = CourseExamConfiguration.get_configuration_for_course(course_id) 

38 if not course_configuration or course_configuration.provider is None: 

39 log.info('Forwarding request to legacy edx-proctoring service for course %s', course_id) 

40 return legacy_view.as_view()(request=request, *view_args, **view_kwargs) 

41 

42 return None
